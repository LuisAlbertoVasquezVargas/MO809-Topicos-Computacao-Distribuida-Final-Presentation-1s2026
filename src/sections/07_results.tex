        \item \textbf{Software Stack:} 
        \begin{itemize}
          \item \textbf{OS:} Ubuntu 14.04 (64-bit).
          \item \textbf{Frameworks:} Spark 2.1.1, Hadoop 2.6.0.

        \vfill
        \item \textbf{Memory Mode:} 
        \begin{itemize}
          \item \textbf{Level:} \texttt{MEMORY\_ONLY}.
          \item \textbf{Strategy:} RDDs are cached in RAM.
          \item \textbf{Impact:} Bypasses Disk I/O bottlenecks during $L_k$ generation.

\begin{frame}{Experimental Results: Performance Analysis}
    \begin{columns}[onlytextwidth, T]
        \column{0.5\textwidth}
        \begin{itemize}
            \item \textbf{Observation 1: Hash Tree (HT) Bottleneck}
            \begin{itemize}
                \item High collision rates and linear search hinder performance.
            \end{itemize}
            \item \textbf{Observation 2: Trie \& HTT Efficiency}
            \begin{itemize}
                \item Significant \textbf{8x speedup} observed on BMS datasets.
            \end{itemize}
            \item \textbf{Observation 3: Spark Advantage}
            \begin{itemize}
                \item Hash Table Trie (HTT) architecture minimizes "cache penalties."
            \end{itemize}
        \end{itemize}
