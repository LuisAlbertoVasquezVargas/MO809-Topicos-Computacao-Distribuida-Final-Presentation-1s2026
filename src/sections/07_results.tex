\section{Experimental Results}
\begin{frame}{Experimental Results: Benchmark Setup}
    \begin{itemize}
        \item \textbf{Cluster Environment:} 
        \begin{itemize}
            \item Apache Spark 2.x cluster.
            \item RDDs persisted in-memory (Storage Level: MEMORY\_ONLY).
        \end{itemize}
        \vfill
        \item \textbf{Datasets:} 
        \begin{itemize}
            \item \textbf{Retail:} Sparse, long transactions.
            \item \textbf{T10I4D100K:} Standard synthetic benchmark for Apriori.
            \item \textbf{Mushroom/Chess:} Dense datasets with high candidate overlap.
        \end{itemize}
        \vfill
        \item \textbf{Metric:} Execution time (seconds) per iteration ($L_k$ generation).
    \end{itemize}
\end{frame}

\begin{frame}{Experimental Results: Performance Analysis}
    \begin{itemize}
        \item \textbf{Observation 1: Hash Tree Bottleneck}
        \begin{itemize}
            \item Consistently slower due to hash collisions and leaf-node linear searches.
            \item Performance degrades sharply as \textit{minsup} decreases (more candidates).
        \end{itemize}
        \vfill
        \item \textbf{Observation 2: Trie Efficiency}
        \begin{itemize}
            \item Significant speedup over Hash Trees due to prefix pruning.
            \item CPU usage is lower, but still limited by sibling-node traversal in dense layers.
        \end{itemize}
        \vfill
        \item \textbf{Observation 3: Hash Table Trie (HTT) Dominance}
        \begin{itemize}
            \item \textbf{Peak Performance:} HTT shows the lowest execution time across all datasets.
            \item \textbf{Scalability:} As candidate counts reach millions, HTT maintains $O(1)$ child lookup, outperforming the standard Trie by 15--20\%.
        \end{itemize}
    \end{itemize}
\end{frame}
