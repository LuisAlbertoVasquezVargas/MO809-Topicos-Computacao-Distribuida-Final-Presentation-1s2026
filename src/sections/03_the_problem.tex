\section{The FIM Problem}

% --- Slide 1: Origins & Definitions ---
\begin{frame}{Origins and Definition}
    \begin{itemize}
        \item \textbf{Market Basket Analysis:} The historical root of FIM. 
        \item \textbf{The Item Base:} Let $\mathcal{B} = \{i_1, \dots, i_n\}$ be the set of all possible items.
        \item \textbf{Itemsets:} Any subset $I \subseteq \mathcal{B}$.
        \item \textbf{Transactions:} A database $\mathcal{T} = (t_1, \dots, t_m)$ where each transaction $t_k$ is an itemset.
    \end{itemize}

    \vfill

    \begin{alertblock}{Frequent Itemset Mining (FIM)}
        The task of discovering all itemsets $I$ that appear in the database $\mathcal{T}$ with a frequency (support) greater than or equal to a threshold $s_{min}$:
        \[ s_{\mathcal{T}}(I) = |\{k \in \{1, \dots, m\} \mid I \subseteq t_k\}| \geq s_{min} \]
    \end{alertblock}
\end{frame}

% --- Slide 3: The Antimonotonicity Rule ---
\begin{frame}{The Antimonotonicity Rule}
    \begin{itemize}
        \item \textbf{The Core Principle:} The support of an itemset can never exceed the support of its subsets.
        \item \textbf{Formal Rule:} $\forall I, J: I \subseteq J \implies s_{\mathcal{T}}(I) \geq s_{\mathcal{T}}(J)$.
    \end{itemize}

    \vfill

    \begin{alertblock}{The Pruning (Downward Closure) Property}
        \centering
        If an itemset is \textbf{infrequent}, all of its supersets are also \textbf{infrequent}.
    \end{alertblock}

    \vfill

    \begin{itemize}
        \item \textbf{Search Strategy:} We "cut" the search lattice. If a node fails $s_{min}$, its entire downward branch is discarded.
        \item This defines a spatial boundary in the lattice: the \textbf{Frequent Border}.
    \end{itemize}
\end{frame}

% --- Slide 6: The Apriori Algorithm (The Strategy) ---
\begin{frame}{The Apriori Algorithm}
    Apriori is a \textbf{level-wise} search algorithm. It uses a "generate-and-test" strategy:
    
    \begin{enumerate}
        \item \textbf{K=1:} Find all frequent items (e.g., $\{a\}, \{b\}$).
        \item \textbf{Join Step:} Generate candidates of size $k+1$ from frequent $k$-itemsets.
        \item \textbf{Prune Step:} Use the \textbf{Antimonotonicity Rule} to delete invalid candidates.
        \item \textbf{Count Step:} Scan the database to calculate actual support.
    \end{enumerate}

    \vfill
    \begin{center}
        \textit{Repeat until no more candidates can be generated.}
    \end{center}
\end{frame}

% --- Slide 7: The Join Step (Candidate Generation) ---
\begin{frame}{The Join Step: $L_k \to C_{k+1}$}
    \begin{alertblock}{The Prefix-Join Rule}
        Join frequent itemsets $p, q \in L_k$ \textbf{only if} they share the same prefix of length $k-1$:
        \[ p = \{i_1, \dots, i_{k-1}, i_k\} \quad q = \{i_1, \dots, i_{k-1}, i_{k'}\} \]
        \[ \text{Candidate } c = \{i_1, \dots, i_{k-1}, i_k, i_{k'}\} \]
    \end{alertblock}

    \vfill

    \textbf{Why this works:}
    \begin{itemize}
        \item It generates the \textit{minimum} necessary candidates.
        \item Example: If $\{a, c\}$ and $\{a, d\}$ are frequent, $\{a, c, d\}$ is a potential candidate.
        \item If $\{a, c\}$ is frequent but $\{c, d\}$ is \textbf{not}, the join logic naturally skips $\{a, c, d\}$.
    \end{itemize}
\end{frame}

% --- Slide 8: The Computational Bottleneck ---
\begin{frame}{Why is this slow in practice?}
    \textbf{The Problem: Searching a Giant Haystack}
    \begin{itemize}
        \item Even with pruning, we might have \textbf{millions} of candidate itemsets.
        \item For every single transaction, we must check which candidates are present.
    \end{itemize}

    \vfill

    \begin{alertblock}{The "Librarian" Analogy}
        Imagine looking for a specific book in a massive library:
        \begin{itemize}
            \item \textbf{Standard Search:} Like walking down 10 different hallways just to find out a book isn't there.
            \item \textbf{The Efficiency Gap:} We spend 90\% of our time "walking" (moving through memory) and only 10\% actually "reading" (counting frequency).
        \end{itemize}
    \end{alertblock}
