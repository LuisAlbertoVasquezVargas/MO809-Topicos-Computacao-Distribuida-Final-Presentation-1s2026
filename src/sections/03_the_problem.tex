\section{The FIM Problem}

% --- Slide 1: Origins & Definitions ---
\begin{frame}{Origins and Definition}
    \begin{itemize}
        \item \textbf{Market Basket Analysis:} The historical root of FIM. 
        \item \textbf{The Item Base:} Let $\mathcal{B} = \{i_1, \dots, i_n\}$ be the set of all possible items.
        \item \textbf{Itemsets:} Any subset $I \subseteq \mathcal{B}$.
        \item \textbf{Transactions:} A database $\mathcal{T} = (t_1, \dots, t_m)$ where each transaction $t_k$ is an itemset.
    \end{itemize}

    \vfill

    \begin{alertblock}{Frequent Itemset Mining (FIM)}
        The task of discovering all itemsets $I$ that appear in the database $\mathcal{T}$ with a frequency (support) greater than or equal to a threshold $s_{min}$:
        \[ s_{\mathcal{T}}(I) = |\{k \in \{1, \dots, m\} \mid I \subseteq t_k\}| \geq s_{min} \]
    \end{alertblock}
\end{frame}

% --- Slide 3: The Antimonotonicity Rule ---
\begin{frame}{The Antimonotonicity Rule}
    \begin{itemize}
        \item \textbf{The Core Principle:} The support of an itemset can never exceed the support of its subsets.
        \item \textbf{Formal Rule:} $\forall I, J: I \subseteq J \implies s_{\mathcal{T}}(I) \geq s_{\mathcal{T}}(J)$.
    \end{itemize}

    \vfill

    \begin{alertblock}{The Pruning (Downward Closure) Property}
        \centering
        If an itemset is \textbf{infrequent}, all of its supersets are also \textbf{infrequent}.
    \end{alertblock}

    \vfill

    \begin{itemize}
        \item \textbf{Search Strategy:} We "cut" the search lattice. If a node fails $s_{min}$, its entire downward branch is discarded.
        \item This defines a spatial boundary in the lattice: the \textbf{Frequent Border}.
    \end{itemize}
\end{frame}

% --- Slide 6: The Bottleneck ---
\begin{frame}{The Computational Bottleneck}
    \begin{itemize}
        \item \textbf{Candidate Explosion:} Even with pruning, $C_k$ can reach millions in massive datasets.
        \item \textbf{Frequency Counting:} The most expensive step. Identifying which candidates are present in each transaction.
    \end{itemize}

    \vfill

    \begin{alertblock}{Limitations of Standard Structures}
        \begin{enumerate}
            \item \textbf{Memory Locality:} Excessive pointer hopping.
            \item \textbf{Traversal Overhead:} Deep, empty branches in sparse data.
            \item \textbf{Latency:} $O(\log n)$ lookup vs the required $O(1)$.
        \end{enumerate}
    \end{alertblock}
    
    \vfill
    \centering
    \textit{This motivates the need for the \textbf{Hash Table Trie (HTT)}.}
\end{frame}
