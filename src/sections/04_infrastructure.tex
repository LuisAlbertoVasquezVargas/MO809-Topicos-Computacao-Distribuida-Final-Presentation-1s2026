% --- Slide 2: The Evolution of Speed ---
\begin{frame}{From the "Basement" to the "Desk"}
    The evolution of Big Data is the story of moving data closer to the CPU.

    \begin{itemize}
        \item \textbf{Hadoop Era (2006): The "Basement" (Disk)}
            \begin{itemize}
                \item Data lived on Hard Drives.
                \item \textbf{Latency:} Like a librarian walking to the basement for every book.
            \end{itemize}
        \vfill
        \item \textbf{Spark Era (2014): The "Desk" (RAM)}
            \begin{itemize}
                \item RAM became a commodity. Data now lives in Memory.
                \item \textbf{Efficiency:} The librarian keeps all the books right on the desk.
            \end{itemize}
    \end{itemize}

    \vfill
    \begin{table}
        \centering
        \footnotesize
        \begin{tabular}{|l|c|c|}
            \hline
            \textbf{Feature} & \textbf{Hadoop (Disk)} & \textbf{Spark (RAM)} \\ \hline
            RAM Cost (approx) & \$10 / MB & \textbf{\$0.01 / MB} \\ \hline
            Processing Speed & 1x & \textbf{100x Faster} \\ \hline
            Data Access & Sequential/Slow & \textbf{Random/Instant} \\ \hline
        \end{tabular}
    \end{table}
\end{frame}

% --- Slide 3: Spark RDDs and Broadcasting ---
\begin{frame}{Apache Spark Architecture}
    \begin{itemize}
        \item \textbf{RDD (Resilient Distributed Dataset):}
            \begin{itemize}
                \item Database is partitioned across the cluster's RAM.
            \end{itemize}
        \item \textbf{The Broadcast Pattern:}
            \begin{itemize}
                \item The Driver node generates the candidates ($C_k$).
                \item It \textbf{broadcasts} them once to every worker's RAM.
            \end{itemize}
    \end{itemize}

    \vfill
    \begin{alertblock}{The New Bottleneck}
        Now that data is on the "Desk" (RAM), the disk is no longer the problem. The bottleneck shifts to the \textbf{CPU} trying to search through millions of candidates.
    \end{alertblock}
\end{frame}

% --- Slide 4: The Traditional Trie (The "Walk") ---
\begin{frame}{The Traditional Trie (Pointer-Based)}
    \begin{columns}
        \begin{column}{0.6\textwidth}
            \begin{itemize}
                \item \textbf{Mechanism:} Nodes are linked by pointers.
                \item \textbf{The Scan:} CPU checks children one by one.
                \item \textbf{The Penalty:} 
                    \begin{itemize}
                        \item $O(\text{branching factor})$ search.
                        \item \textbf{Cache Misses:} CPU "stutters" while waiting.
                    \end{itemize}
            \end{itemize}
        \end{column}
        \begin{column}{0.4\textwidth}
            \centering
            \begin{tikzpicture}[scale=0.8, every node/.style={draw,circle,font=\tiny}]
                \node {root} 
                    child {node {a}} 
                    child {node {b}} 
                    child {node {c} child {node {d}}};
            \end{tikzpicture}
        \end{column}
    \end{columns}
\end{frame}

% --- Slide 5: The Solution: Hash Table Trie (HTT) ---
\begin{frame}{The "Teleporter": Hash Table Trie}
    \begin{block}{The Mechanism}
        Every node contains a \textbf{Hash Table} instead of a list.
        \begin{itemize}
            \item \textbf{Key:} Item ID $\to$ \textbf{Value:} Pointer to Child.
        \end{itemize}
    \end{block}

    \vfill
    \begin{itemize}
        \item \textbf{Complexity:} Reduced from $O(b)$ to \textbf{$O(1)$}.
        \item \textbf{The "Teleport":} Use the Item ID to jump directly to data.
        \item \textbf{Result:} Maximize CPU efficiency; no idling during lookups.
    \end{itemize}
\end{frame}

    \end{alertblock}
