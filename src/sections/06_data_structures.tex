% --- Slide 3: The HTT Solution ---
\begin{frame}[t]{The Hash Table Trie (HTT) Hybrid}
    \vspace{-0.2cm}
    \textbf{The Innovation: Localized Map Lookup}
    
    \begin{center}
    \begin{tikzpicture}[scale=0.7, every node/.style={transform shape}]
        \node[draw, circle, thick] (P) at (0,0) {Parent Node};
        \node[draw, circle, thick, fill=red!5] (C) at (5,0) {Child Node $i$};
        
        \draw[dashed, gray] (-1.2,-1.2) rectangle (1.2,1.2);
        \node[draw, rectangle, fill=red!10, minimum width=3.5cm, align=center, font=\scriptsize] (HT) at (0.8,-2) {Local Hash Table\\ \texttt{ItemID $\rightarrow$ ChildPointer}};
        
        \draw[->, ultra thick, red, >=stealth] (P) -- (C) node[midway, above] {\small $O(1)$ Lookup};
        \draw[->, thin, gray, dashed] (0,-0.8) -- (HT);
    \end{tikzpicture}
    \end{center}

    \vspace{-0.4cm}
    \begin{table}[]
        \centering
        \scriptsize
        \begin{tabular}{l|c|c|c}
            \toprule
            \textbf{Metric} & \textbf{Hash Tree (HT)} & \textbf{Standard Trie} & \textbf{HTT (Proposed)} \\ \midrule
            Sibling Lookup & $O(n)$ linear scan & $O(b)$ search & $\mathbf{O(1)}$ \textbf{lookup} \\ 
            Traversal Complexity & $O(k \cdot d)$ & $O(k \cdot b)$ & $\mathbf{O(k)}$ \\ \bottomrule
        \end{tabular}
    \end{table}

    \begin{columns}[T]
        \begin{column}{0.45\textwidth}
            \begin{alertblock}{Notation Reference}
                \scriptsize
                \begin{itemize}
                    \item \textbf{$k$}: Direct traversal steps.
                    \item \textbf{$O(1)$}: Hash-mapped child access.
                \end{itemize}
            \end{alertblock}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{alertblock}{Technical Verdict}
                \scriptsize
                Complexity: \textbf{$O(k)$}. By embedding hash tables in nodes, HTT eliminates $b$ and $d$ overhead, achieving true linear performance.
            \end{alertblock}
        \end{column}
    \end{columns}
\end{frame}
