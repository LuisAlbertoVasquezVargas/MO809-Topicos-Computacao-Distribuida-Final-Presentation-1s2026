\section{Introduction \& Motivation}
\begin{frame}{Introduction \& Motivation}
    \begin{itemize}
        \item \textbf{The Goal:} Discovering hidden patterns in massive transactional datasets.
        \item \textbf{The Barrier: Combinatorial Explosion}
        \begin{itemize}
            \item For a set of $n$ items, the search space is $O(2^n)$.
            \item If $n=100$: $2^{100} \approx 1.26 \times 10^{30}$ combinations.
            \item \textbf{Brute Force:} Checking every subset would take longer than the age of the universe.
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Introduction \& Motivation}
    \begin{itemize}
        \item \textbf{Hardware Evolution:} 
        \begin{itemize}
            \item We have the ``Ferrari'' of computing: \textbf{Apache Spark RDDs}.
        \end{itemize}
        \vfill
        \item \textbf{Software Bottleneck:} 
        \begin{itemize}
            \item Legacy structures (Hash Trees) create a \textbf{traffic jam} that wastes the speed of RAM.
        \end{itemize}
    \end{itemize}
    \vfill
    \begin{alertblock}{The Core Thesis}
        Infrastructure alone cannot solve $O(2^n)$. To exploit the speed of Spark, we must replace ``Disk-era'' structures with memory-optimized \textbf{Tries}.
    \end{alertblock}
\end{frame}

% --- SECTION 2: THE FIM PROBLEM ---
\section{The FIM Problem}
\begin{frame}{The FIM Problem}
    \begin{itemize}
        \item \textbf{Frequent Itemset Mining (FIM):} Finding subsets with frequency $\geq$ \textit{minsup}.
        \item \textbf{The ``Bundle Deal'' Analogy:} 
        \begin{itemize}
            \item If customers buy GPUs, what else do they buy?
            \item $n=1000$ items lead to $2^{1000}$ possible combinations.
        \end{itemize}
        \vfill
        \item \textbf{The Apriori Principle:} 
        \begin{quote}
            ``Any subset of a frequent itemset must also be frequent.''
        \end{quote}
        \item We prune the space, but we need efficient storage for millions of candidates.
    \end{itemize}
\end{frame}

\section{Conclusion}
\begin{frame}{Conclusion: The Sequential Context}
    \begin{itemize}
        \item \textbf{Theoretical Expectation:} 
        \begin{itemize}
            \item The \textbf{Hash Table Trie} was designed to accelerate itemset traversal by combining Trie logic with hashing.
        \end{itemize}
        \vfill
        \item \textbf{Previous Practical Failure:} 
        \begin{itemize}
            \item In earlier sequential implementations, where \textbf{memory was a strict bottleneck}, the hashing overhead caused it to underperform.
            \item It was often documented as the least efficient variant in non-distributed settings.
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Conclusion: The Spark Paradigm Shift}
    \begin{itemize}
        \item \textbf{Distributed Advantage:} 
        \begin{itemize}
            \item Within Spark's \textbf{distributed memory (RDD)} architecture, the Hash Table Trie finally realizes its theoretical efficiency.
        \end{itemize}
        \vfill
        \item \textbf{Benchmark Results:} 
        \begin{itemize}
            \item Experimental data shows it performs \textbf{significantly better} than the traditional Hash Tree.
            \item It maintains a slight edge over the standard Trie across most datasets.
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Conclusion: Final Verdict}
    \begin{itemize}
        \item \textbf{In-Memory Optimization:} 
        \begin{itemize}
            \item Memory-optimized structures are the key to high-performance FIM on Spark.
        \end{itemize}
        \vfill
        \item \textbf{Summary:}
        \begin{itemize}
            \item The \textbf{Hash Table Trie}, once considered the ``worst'' in sequential settings, is now the most robust choice for RDD-based Apriori.
            \item Infrastructure alone (Spark) isn't enough; the underlying \textbf{data structure} must evolve to match the hardware.
        \end{itemize}
    \end{itemize}
\end{frame}
