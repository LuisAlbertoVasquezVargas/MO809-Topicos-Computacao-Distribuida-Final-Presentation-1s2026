\section{Informal Model}

\begin{frame}{The "Bundle Deal" Strategy}
    Imagine a shop where we want to create popular \textbf{Bundle Deals}.
    \vfill
    \begin{itemize}
        \item \textbf{Receipts:} Each list represents one customer visit.
        \item \textbf{Rule:} Customers buy at most \textit{one} of each item (e.g., Bread, Milk, Tea).
        \item \textbf{Popularity Threshold ($s\%$ or $minsup$):} We only care about bundles appearing in at least $s\%$ of total receipts.
    \end{itemize}
    \vfill
    \begin{block}{Full Item Inventory}
        $\{\text{Milk, Bread, Tea, Coffee, Soda, Juice, Butter, Cookies}\}$
    \end{block}
\end{frame}

\begin{frame}{The "Shortcut" Logic}
    \begin{itemize}
        \item If \textbf{Coffee} is in 5\% of receipts but our \textbf{Threshold} is 50\%:
        \item \textbf{Pruning:} The bundle \textbf{\{Coffee + Milk\}} cannot be popular.
        \item \textbf{Efficiency:} We save computation by ignoring all bundles containing Coffee.
    \end{itemize}
    \vfill
\end{frame}

\section{Formal Model}

\begin{frame}{Formal Mathematical Model}
    Let $I = \{i_1, i_2, \dots, i_m\}$ be the set of items.
    \begin{definition}
        A \textbf{Transaction Database} $D$ is a set of receipts $T \subseteq I$.
    \end{definition}
    \vfill
    The goal is to find all $X \subseteq I$ such that:
    \begin{equation*}
        \text{support}(X) = \frac{|\{T \in D \mid X \subseteq T\}|}{|D|} \ge \sigma
    \end{equation*}
    where $\sigma$ is the \textbf{Minimum Support} ($minsup$).
\end{frame}

\begin{frame}{The Apriori Property}
    The algorithm relies on \textbf{Down-closure Monotonicity}:
    \vfill
    \begin{exampleblock}{Theorem}
        For any itemsets $X$ and $Y$:
        \[ (X \subseteq Y) \implies \text{support}(X) \ge \text{support}(Y) \]
    \end{exampleblock}
    \vfill
    \textbf{Consequence:} If $X$ is infrequent, all its supersets $Y$ are also infrequent, allowing for massive search space reduction.
\end{frame}

\section{The Distributed Challenge}

\begin{frame}{The Spark Bottleneck}
    In a distributed environment like Spark, receipts are spread across nodes.
    \vfill
    \begin{itemize}
        \item \textbf{Candidate Generation:} Millions of potential bundles are created.
        \item \textbf{The Wall:} For every local receipt, the worker must check: 
        \textit{"Which candidates match this specific receipt?"}
        \item \textbf{The Core Problem:} Standard search is too slow; the performance is bound by the \textbf{efficiency of the matching data structure}.
    \end{itemize}
\end{frame}
